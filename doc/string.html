Method unhyphenate

Replaces all hyphens with spaces

  • obj must be a String.
    (String to unhypenate)

  • caps must be a Boolean.
    ([Optional] True to capitalize each word)

(Unhyphenated String)

unhyphenate : function ( obj, caps ) { caps = ( caps === true ); var result = ""; if ( obj.indexOf( "-" ) > -1 ) { array.each( string.trim( obj ).split( "-" ), function ( i ) { result += ( caps ? string.capitalize( i ) : i ) + " "; }); } else { result = caps ? string.capitalize( obj ) : obj; } return string.trim( result ); } };
